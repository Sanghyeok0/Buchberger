\section*{Gr\"obner Bases and the Extension Theorem}

Let $k$ be a field and let $f \in k[x_1,\dots,x_n]$ be nonzero.
We can write
\begin{equation}\label{eq:leading-in-x1}
  f = c_f(x_2,\dots,x_n)\,x_1^N
      + \text{terms in which $x_1$ has degree $< N$},
\end{equation}
where $N \ge 0$ and $c_f \in k[x_2,\dots,x_n]$ is nonzero.
We define
\[
  \deg(f,x_1) := N
\]
and we set $c_f := 0$ when $f = 0$.

\begin{lemma}\label{lem:lex-standard-representation}
Assume that
\[
  f = \sum_{j=1}^t A_j g_j
\]
is a standard representation for the lex order with
$x_1 > \cdots > x_n$. Then:
\begin{enumerate}[label=(\roman*)]
  \item $\deg(f,x_1) \ge \deg(A_j g_j,x_1)$ whenever $A_j g_j \ne 0$.
  \item $c_f = \displaystyle
        \sum_{\deg(A_j g_j,x_1)=N} c_{A_j}\,c_{g_j}$,
        where $N = \deg(f,x_1)$.
\end{enumerate}
\end{lemma}
